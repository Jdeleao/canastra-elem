<script>

(() => {

  // Utils

  function sleep(ms) { return new Promise(resolve => setTimeout(resolve, ms)); }

 

  // Classes

 

  class Carta {

    constructor(tipo, naipe = null, valor = null) {

      this.tipo = tipo; // "normal", "Fogo", "Água", "Ar", "Pedra"

      this.naipe = naipe;

      this.valor = valor;

    }

    nome() {

      if (this.tipo === 'normal') {

        const naipesSimbolos = { 'paus': '♣', 'copas': '♥', 'espadas': '♠', 'ouros': '♦' };

        let v = this.valor;

        switch(v) {

          case 1: v = 'A'; break;

          case 11: v = 'J'; break;

          case 12: v = 'Q'; break;

          case 13: v = 'K'; break;

          default: v = v.toString();

        }

        return `${v}${naipesSimbolos[this.naipe]}`;

      } else {

        return this.tipo;

      }

    }

    cssClass() {

      return `carta ${this.tipo}`;

    }

  }

 

  class Jogador {

    constructor(nome) {

      this.nome = nome;

      this.mao = [];

      this.jogos = [];

      this.naipe_trancado = new Set();

      this.protecao = new Set(); // proteção contra Fogo/Água

      this.efeitos_ar = new Set(); // efeitos acumulados de Ar

    }

    podeUsarCarta(carta) {

      if (carta.tipo !== 'normal') return true;

      return !this.naipe_trancado.has(carta.naipe);

    }

  }

 

  class CanastraJogo {

    constructor() {

      this.jogadores = [new Jogador("Jogador 1"), new Jogador("Jogador 2")];

      this.baralho = [];

      this.monteDescarte = [];

      this.jogadorAtualIndex = 0;

      this.jogoFinalizado = false;

      this.mensagemCallbacks = [];

      this.selecaoCartas = [];

      this.estadoFormarJogo = false;

      this.uiAtualizar = () => {};

    }

 

    log(texto) {

      this.mensagemCallbacks.forEach(cb => cb(texto));

    }

 

    onLog(cb) {

      this.mensagemCallbacks.push(cb);

    }

 

    gerarBaralho() {

      const naipes = ['paus', 'copas', 'espadas', 'ouros'];

      this.baralho = [];

      for (let naipe of naipes) {

        for (let valor = 1; valor <= 13; valor++) {

          this.baralho.push(new Carta('normal', naipe, valor));

        }

      }

      for (let i = 0; i < 2; i++) {

        this.baralho.push(new Carta('Fogo'));

        this.baralho.push(new Carta('Água'));

        this.baralho.push(new Carta('Ar'));

        this.baralho.push(new Carta('Pedra'));

      }

      this.baralho = this.shuffle(this.baralho);

    }

 

    shuffle(array) {

      for (let i = array.length - 1; i > 0; i--) {

        let j = Math.floor(Math.random() * (i + 1));

        [array[i], array[j]] = [array[j], array[i]];

      }

      return array;

    }

 

    distribuirCartas() {

      for (let i = 0; i < 11; i++) {

        this.jogadores.forEach(jog => {

          const carta = this.baralho.pop();

          jog.mao.push(carta);

        });

      }

    }

 

    iniciar() {

      this.gerarBaralho();

      this.distribuirCartas();

      this.monteDescarte = [];

      this.jogoFinalizado = false;

      this.jogadorAtualIndex = 0;

      this.selecaoCartas = [];

      this.estadoFormarJogo = false;

      this.log(`=== Jogo iniciado! Vez de ${this.jogadorAtual.nome} ===`);

      this.uiAtualizar();

    }

 

    get jogadorAtual() {

      return this.jogadores[this.jogadorAtualIndex];

    }

 

    proximoTurno() {

      this.selecaoCartas = [];

      this.estadoFormarJogo = false;

 

      // Efeitos de Ar: limpar efeitos

      this.jogadorAtual.efeitos_ar.clear();

 

      // Proteção expira

      this.jogadorAtual.protecao.clear();

 

      // Próximo jogador

      this.jogadorAtualIndex = (this.jogadorAtualIndex + 1) % this.jogadores.length;

      this.log(`\n=== Vez de ${this.jogadorAtual.nome} ===`);

      this.uiAtualizar();

    }

 

    comprarCarta(fonte) {

      const jogador = this.jogadorAtual;

      let carta;

      if (fonte === 'baralho') {

        if (this.baralho.length === 0) {

          this.log("Baralho acabou, jogo termina.");

          this.jogoFinalizado = true;

          this.uiAtualizar();

          return false;

        }

        carta = this.baralho[this.baralho.length - 1];

        if (carta.tipo === 'normal' && jogador.naipe_trancado.has(carta.naipe)) {

          this.log(`Não pode comprar do baralho: naipe ${carta.naipe} trancado.`);

          return false;

        }

        carta = this.baralho.pop();

        jogador.mao.push(carta);

        this.log(`${jogador.nome} comprou ${carta.nome()} do baralho.`);

        this.uiAtualizar();

        return true;

      } else if (fonte === 'descarte') {

        if (this.monteDescarte.length === 0) {

          this.log("Monte de descarte vazio.");

          return false;

        }

        carta = this.monteDescarte[this.monteDescarte.length - 1];

        if (carta.tipo === 'normal' && jogador.naipe_trancado.has(carta.naipe)) {

          this.log(`Não pode comprar do monte: naipe ${carta.naipe} trancado.`);

          return false;

        }

        carta = this.monteDescarte.pop();

        jogador.mao.push(carta);

        this.log(`${jogador.nome} comprou ${carta.nome()} do monte de descarte.`);

        this.uiAtualizar();

        return true;

      }

      return false;

    }

 

    descartarCarta(index) {

      const jogador = this.jogadorAtual;

      if (index < 0 || index >= jogador.mao.length) {

        this.log("Índice inválido para descartar.");

        return false;

      }

      const carta = jogador.mao[index];

      jogador.mao.splice(index, 1);

      this.monteDescarte.push(carta);

      this.log(`${jogador.nome} descartou ${carta.nome()}.`);

      this.proximoTurno();

      return true;

    }

 

    // Verifica se é uma trinca/quádrupla válida

    ehTrinca(cartas) {

      if (cartas.length < 3) return false;

      const valores = new Set(cartas.map(c => c.valor));

      return valores.size === 1;

    }

 

    // Verifica se é uma sequência válida (mesmo naipe, valores consecutivos)

    ehSequencia(cartas) {

      if (cartas.length < 3) return false;

      const porNaipe = new Set(cartas.map(c => c.naipe));

      if (porNaipe.size !== 1) return false;

 

      const ordenado = [...cartas].sort((a, b) => a.valor - b.valor);

      for (let i = 1; i < ordenado.length; i++) {

        if (ordenado[i].valor !== ordenado[i-1].valor + 1) {

          return false;

        }

      }

      return true;

    }

 

    formarJogo(indices) {

      const jogador = this.jogadorAtual;

      if (indices.length < 3) {

        this.log("Para formar jogo precisa escolher ao menos 3 cartas.");

        return false;

      }

      let cartasSelecionadas = [];

      for (let i of indices) {

        if (i < 0 || i >= jogador.mao.length) {

          this.log(`Índice ${i+1} inválido.`);

          return false;

        }

        cartasSelecionadas.push(jogador.mao[i]);

      }

 

      // Cartas elementares não entram em jogos

      for (let c of cartasSelecionadas) {

        if (c.tipo !== 'normal') {

          this.log("Cartas elementares não podem formar jogos.");

          return false;

        }

        if (jogador.naipe_trancado.has(c.naipe)) {

          this.log(`Naipe ${c.naipe} está trancado, não pode usar esta carta.`);

          return false;

        }

      }

 

      // Testar trinca ou sequência

      if (this.ehTrinca(cartasSelecionadas)) {

        const valor = cartasSelecionadas[0].valor;

        const naipe = cartasSelecionadas[0].naipe;

        this.log(`${jogador.nome} formou uma trinca de ${valor} (${cartasSelecionadas.length} cartas).`);

        // Remover cartas da mão

        for (let i = indices.sort((a,b)=>b-a)) { // ordem decrescente para não bagunçar índices

          jogador.mao.splice(i, 1);

        }

        this.uiAtualizar();

        return true;

      } else if (this.ehSequencia(cartasSelecionadas)) {

        const naipe = cartasSelecionadas[0].naipe;

        const min = cartasSelecionadas.reduce((a,b) => Math.min(a, b.valor), 13);

        const max = cartasSelecionadas.reduce((a,b) => Math.max(a, b.valor), 1);

        this.log(`${jogador.nome} formou uma sequência de ${min} a ${max} de ${naipe}.`);

        for (let i = indices.sort((a,b)=>b-a)) {

          jogador.mao.splice(i, 1);

        }

        this.uiAtualizar();

        return true;

      } else {

        this.log("As cartas selecionadas não formam um jogo válido (trinca ou sequência).");

        return false;

      }

    }

 

    // Jogar carta elemental

    jogarElemental(index) {

      const jogador = this.jogadorAtual;

      if (index < 0 || index >= jogador.mao.length) {

        this.log("Índice inválido.");

        return false;

      }

      const carta = jogador.mao[index];

      if (carta.tipo === 'normal') {

        this.log("Só pode jogar cartas elementares.");

        return false;

      }

 

      // Aplicar efeito

      switch(carta.tipo) {

        case 'Fogo':

          // Queima o monte de descarte

          if (this.monteDescarte.length > 0) {

            const queimadas = this.monteDescarte.length;

            this.monteDescarte = [];

            this.log(`${jogador.nome} usou Fogo! Queimou ${queimadas} carta(s) do descarte.`);

          } else {

            this.log("Monte de descarte vazio, Fogo não tem efeito.");

          }

          break;

        case 'Água':

          // Permite comprar do monte queimado (não implementado aqui) ou protege

          jogador.protecao.add('Água');

          this.log(`${jogador.nome} usou Água! Está protegido contra Fogo.`);

          break;

        case 'Ar':

          // Pode comprar duas cartas do baralho

          this.comprarCarta('baralho');

          this.comprarCarta('baralho');

          jogador.efeitos_ar.add('ativo');

          this.log(`${jogador.nome} usou Ar! Comprou duas cartas.`);

          break;

        case 'Pedra':

          // Tranca um naipe no oponente

          const oponente = this.jogadores[(this.jogadorAtualIndex + 1) % 2];

          const naipes = ['paus', 'copas', 'espadas', 'ouros'];

          const naipeAleatorio = naipes[Math.floor(Math.random() * naipes.length)];

          oponente.naipe_trancado.add(naipeAleatorio);

          this.log(`${jogador.nome} usou Pedra! Trancou o naipe ${naipeAleatorio} do oponente.`);

          break;

      }

 

      // Descartar a carta elemental

      jogador.mao.splice(index, 1);

      this.monteDescarte.push(carta);

      this.log(`${jogador.nome} descartou ${carta.nome()}.`);

 

      // Próximo turno

      this.proximoTurno();

      return true;

    }

  }

 

  // === INICIALIZAÇÃO DA UI ===

  const jogo = new CanastraJogo();

 

  // Função para atualizar a interface

  jogo.uiAtualizar = function() {

    const estadoDiv = document.getElementById('estado');

    const logDiv = document.getElementById('log');

    const controlsDiv = document.getElementById('controls');

    const descarteViz = document.getElementById('descarte-viz');

 

    // Exibir estado

    estadoDiv.innerHTML = `

      <div class="jogador">

        <strong>${jogo.jogadorAtual.nome}</strong> (${jogo.jogadorAtual.mao.length} cartas)<br>

        Naipes trancados: ${Array.from(jogo.jogadorAtual.naipe_trancado).join(', ') || 'nenhum'}

      </div>

    `;

 

    // Cartas na mão

    const maoDiv = document.createElement('div');

    maoDiv.innerHTML = '';

    jogo.jogadorAtual.mao.forEach((carta, idx) => {

      const span = document.createElement('span');

      span.className = carta.cssClass();

      span.textContent = carta.nome();

      span.onclick = () => {

        // Seleção de cartas

        const index = jogo.selecaoCartas.indexOf(idx);

        if (index === -1) {

          jogo.selecaoCartas.push(idx);

          span.classList.add('selected');

        } else {

          jogo.selecaoCartas.splice(index, 1);

          span.classList.remove('selected');

        }

      };

      span.style.margin = '2px';

      maoDiv.appendChild(span);

    });

 

    // Botões

    controlsDiv.innerHTML = '';

    controlsDiv.appendChild(maoDiv);

 

    const btnComprarBaralho = document.createElement('button');

    btnComprarBaralho.textContent = 'Comprar do Baralho';

    btnComprarBaralho.onclick = () => jogo.comprarCarta('baralho');

 

    const btnComprarDescarte = document.createElement('button');

    btnComprarDescarte.textContent = 'Comprar do Descarte';

    btnComprarDescarte.onclick = () => jogo.comprarCarta('descarte');

 

    const btnFormarJogo = document.createElement('button');

    btnFormarJogo.textContent = 'Formar Jogo com Selecionadas';

    btnFormarJogo.onclick = () => {

      if (jogo.selecaoCartas.length > 0) {

        jogo.formarJogo([...jogo.selecaoCartas]);

        jogo.selecaoCartas = [];

        jogo.uiAtualizar();

      } else {

        alert("Selecione cartas primeiro.");

      }

    };

 

    const btnDescartar = document.createElement('button');

    btnDescartar.textContent = 'Descartar Carta Selecionada';

    btnDescartar.onclick = () => {

      if (jogo.selecaoCartas.length !== 1) {

        alert("Selecione exatamente uma carta para descartar.");

        return;

      }

      jogo.descartarCarta(jogo.selecaoCartas[0]);

    };

 

    const btnJogarElemental = document.createElement('button');

    btnJogarElemental.textContent = 'Usar Carta Elemental';

    btnJogarElemental.onclick = () => {

      if (jogo.selecaoCartas.length !== 1) {

        alert("Selecione uma carta elemental.");

        return;

      }

      const idx = jogo.selecaoCartas[0];

      const carta = jogo.jogadorAtual.mao[idx];

      if (['Fogo', 'Água', 'Ar', 'Pedra'].includes(carta.tipo)) {

        jogo.jogarElemental(idx);

      } else {

        alert("Essa carta não é elemental.");

      }

    };

 

    controlsDiv.appendChild(btnComprarBaralho);

    controlsDiv.appendChild(btnComprarDescarte);

    controlsDiv.appendChild(btnFormarJogo);

    controlsDiv.appendChild(btnDescartar);

    controlsDiv.appendChild(btnJogarElemental);

 

    // Descarte

    if (descarteViz) {

      descarteViz.innerHTML = '';

      if (jogo.monteDescarte.length > 0) {

        const ultima = jogo.monteDescarte[jogo.monteDescarte.length - 1];

        const span = document.createElement('span');

        span.className = ultima.cssClass();

        span.textContent = ultima.nome();

        descarteViz.appendChild(span);

      } else {

        descarteViz.textContent = '(vazio)';

      }

    }

  };

 

  // Log

  jogo.onLog(texto => {

    const logDiv = document.getElementById('log');

    const p = document.createElement('p');

    p.textContent = texto;

    logDiv.appendChild(p);

    logDiv.scrollTop = logDiv.scrollHeight;

  });

 

  // Iniciar jogo

  jogo.iniciar();

 

})();

</script>
